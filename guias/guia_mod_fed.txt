//creamos el workspace Y LA ESTRUCTURA BASE
npx -p @angular/cli@15 ng new amazonmf --create-application=false --package-manager=npm

//con esto generamos el módulo raíz

//configuramos la librería anterior para el módulo shell/ raíz
npx ng add @angular-architects/module-federation@15 --project shell --type host --port 4200
//configuramos la librería anterior para el resto de módulos (remotos/mfe)

npx ng add @angular-architects/module-federation@15 --project perfil --type remote --port 4202
npx ng add @angular-architects/module-federation@15 --project productos --type remote --port 4203

    2.1 CREAMOS UN SERVICIO EN core

        npx ng generate service core/services/communication --project=core

    2.2 CREAMOS UNA INTERFAZ EN models

        npx ng generate interface models/producto-event --project=models

    2.3 DESCRIBIMOS EN SENDAS LIBRERÍAS, QUÉ SE COMPARTE PÚBLICAMENTE 

        EDITAMOS ./projects/models/src/public-api.ts
                 ./projects/models/src/public-api.ts

    2.4 MODIFICAMOS EL TSCONFIG para AÑADIR ESTAS LIBRERÍAS AL BUNDLE DE COMPILACIÓN

         "@models": [
                "./projects/models/src/public-api.ts"
          ]

          "@core-lib": [
                "./projects/models/src/public-api.ts"
          ]

    2.5 REFERENCIAMOS A ESAS LIBRERÍAS DESDE CADA MICRO QUE LO NECESITE

        //esto es lo nuevo, tanto en webpack.config.js del Carrito como de Productos
        sharedMappings:["@core-lib", "@models"] //con esto inyectamos el este MF estas dependencias

    2.6 EN EL MÓDULO DEL SHELL, IMPORTAMOS LA LIBRERÍA DEL CORE 

        sharedMappings:["@core-lib"] //ASÍ CONSEGUIMOS SERVICIOS REALMENTE SINGLETON que no haya duplicidad de servicios y realmente se inyecten las mismas instancias en cada micro

    2.7 Usamos distintos mecanismos de RX/JS para comunicar vía servicio los Micros

        CommunicationService- TRANSMISOR - inyectado en ambos Components
        CarritoComponent- RECEPCIÓN
